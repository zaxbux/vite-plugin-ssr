### Render Control

With `vite-plugin-ssr`, _you_ define how your pages are rendered.

```js
// /pages/index.page.js
// Environment: Node.js, Browser

// We use JSX here but this could also be anything such as a `.vue` or a `.svelte` file

export default Page

function Page() {
  return (
    <>
      <h1>Welcome to vite-plugin-ssr</h1>
      <p>Our first Vite + SSR app.</p>
    </>
  )
}
```

```js
// /pages/_default/_default.page.server.js
// Environment: Node.js

import { html } from 'vite-plugin-ssr'

// This could be any view framework: React 16, React 17, Vue 2, Vue 3, Svelte, Solid, Preact, ...
import { renderToHtml } from 'some-view-framework'

export { render }

async function render(pageContext) {
  // `Page` is the `export default Page` of our `pages/index.page.js` above
  const { Page } = pageContext

  const pageHtml = await renderToHtml(Page)

  const title = 'My first Vite SSR app'

  // We have full control over how our `Page` is rendered to HTML.
  return html`<!DOCTYPE html>
    <html>
      <head>
        <meta charset="UTF-8" />
        <title>${title}</title>
      </head>
      <body>
        <div id="page-root">${html.dangerouslySkipEscape(appHtml)}</div>
      </body>
    </html>`
}
```

```js
// /pages/_default/_default.page.client.js
// Environment: Browser

import { hydrateToDom } from 'some-view-framework'

// This files defines the *entire* browser-side code.
// If we leave this file empty then there is zero browser-side code.

hydrate()

async function hydrate() {
  const pageContext = await getPage()
  const { Page } = pageContext
  await hydrateToDom(Page)
}
```

### Natural Tool Integration

This control means we can easily integrate any tool we want.

Want to use bootstrap or some other CSS framework? Simply follow the CSS's official installation guide.

```js
export async function render(pageContext) {
  // We follow the official Bootstrap installation guide
  return html`<!DOCTYPE html>
    <html>
      <head>
        <link
          href="https://cdn.jsdelivr.net/npm/bootstrap@5.0.2/dist/css/bootstrap.min.css"
          rel="stylesheet"
          integrity="sha384-EVSTQN3/azprG1Anm3QDgpJLIm9Nao0Yz1ztcQTwFspd3yD65VohhpuuCOmLASjC"
          crossorigin="anonymous"
        >
        <title>${title}</title>
      </head>
      <body>
        <div id="page-root">${html.dangerouslySkipEscape(renderToHtml(pageContext.Page))}</div>
        <script
          src="https://cdn.jsdelivr.net/npm/bootstrap@5.0.2/dist/js/bootstrap.bundle.min.js"
          integrity="sha384-MrcW6ZMFYlzcLA8Nl+NtUVF0sA7MsXsP1UyJoMp4YLEuNSfAP+JcXn/tWtIaxVXM"
          crossorigin="anonymous">
        </script>

      </body>
    </html>`
}
```

Want to use some browser-side library such as Google Analytics or Sentry? Simply follow their official installation guide.

```js
// /pages/_default/_default.page.client.js
// Environment: Browser

import { hydrateToDom } from 'some-view-framework'

// This files defines the *entire* browser-side code.
// If we leave this file empty then there is zero browser-side code.

hydrate()

async function hydrate() {
  const pageContext = await getPage()
  const { Page } = pageContext
  await hydrateToDom(Page)
}
```

Same goes for GraphQL, Vuex, Redux, jQuery, Service Workes, ...: we can simply follow their official installation guide.

In other words, **integration is natural**.
